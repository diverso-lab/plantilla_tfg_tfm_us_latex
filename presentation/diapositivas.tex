\begin{frame}{Contexto y motivación}
  % (Espacio reservado para una figura que sitúe el dominio del problema)
  \ph{Sitúa el dominio en 2--3 ideas: quién sufre el problema, por qué
  importa ahora y qué coste tiene no resolverlo. Apóyate en una figura,

\end{frame}

  2--3 evidencias o carencias de las soluciones actuales.}
\end{frame}
